// Stack implementation using class
class Stack {
    private:
    int[] data;
    int top;
    int capacity;

    public:
    Stack(int cap) {
        this.capacity = cap;
        this.top = -1;
        this.data = {};
    }

    void push(int val) {
        if (this.top < this.capacity - 1) {
            this.top = this.top + 1;
            print(val);
        } else {
            print("Stack overflow");
        }
    }

    int pop() {
        if (this.top >= 0) {
            this.top = this.top - 1;
            return this.top + 1;
        }
        print("Stack underflow");
        return -1;
    }

    bool isEmpty() {
        return this.top == -1;
    }

    int size() {
        return this.top + 1;
    }
}

Stack s = new Stack(10);
s.push(1);
s.push(2);
s.push(3);
print(s.size());
print(s.isEmpty());
